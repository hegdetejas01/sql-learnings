### DML


CREATE DATABASE sql_learning
CREATE TABLE users(
    user_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL
)

1. SELECT * FROM sql_learning.smartphones WHERE 1

# WHERE 1 is optional

2. SELECT * FROM sql_learning.users

3. Filter Columns:

    SELECT model, price, rating, brand_name FROM sql_learning.smartphones 
    # gets only the said columns - model, price, rating, brand_name

    SELECT model, battery_capacity, os FROM sql_learning.smartphones WHERE 1

4. Renaming the columns - alisaing
    SELECT model, battery_capacity AS 'mAh' , os AS 'Operating System' FROM sql_learning.smartphones WHERE 1

5. Creating mathematical expression inside a query
    Eg: Squaring resolution width 
    SELECT model, resolution_width*resolution_width FROM sql_learning.smartphones

    Eg: Eg: Squaring resolution height adding it to square of resolution width 
    SELECT model, resolution_width*resolution_width + resolution_height*resolution_height 
    FROM sql_learning.smartphones

    Eg: Calculating PPI using resolution width, resolution height, screen size
    SELECT 
    model, 
    SQRT(resolution_width*resolution_width + resolution_height*resolution_height)/screen_size AS 'PPI' 
    FROM sql_learning.smartphones

    SELECT model, rating/10 FROM sql_learning.smartphones


6. Creating Constants
    * This code creates a new column called type and the value will be 'smartphone' (VIEW, not a parmanent change)
    SELECT  model, 'smartphone' AS 'type' FROM sql_learning.smartphones

7. Disntict - fetching unique values
    * to fetch the unique brands from data
    SELECT DISTINCT(brand_name) as 'ALL BRANDS' FROM sql_learning.smartphones

    * brands in processor brand
    SELECT DISTINCT(processor_brand) as 'ALL PROCESSORS' FROM sql_learning.smartphones

8. Unique Combinations
    * fetch unique combination of brand_name and processor_brand
    SELECT DISTINCT brand_name, processor_brand FROM sql_learning.smartphones

10. Between operator - gives data between a given range

11. IN and NOT IN
# get the phones where processor_brand is snapdragon or bionic or dimensity
    SELECT * FROM sql_learning.smartphones 
    WHERE processor_brand = 'snapdragon' OR 
    processor_brand = 'bionic' OR 
    processor_brand = 'dimensity'

    ## OR ##

    SELECT * FROM sql_learning.smartphones 
    WHERE processor_brand IN ('snapdragon','bionic','dimensity')

# phones not with processor_brand is snapdragon or bionic or dimensity
    SELECT * FROM sql_learning.smartphones 
    WHERE processor_brand NOT IN ('snapdragon','bionic','dimensity')
