### RANK - Gives ranking for partition
* the same rank will be given for equal items Eg. if 2 students have marks 25, both will be ranked 3
* but since 3 is used 2 times, the next rank will be 5 and not 4 [it considers 3 and 4 are used, but donot display it]

# give rank to students 
    USE sql_learning;
    SELECT * , RANK() OVER(ORDER BY marks DESC) FROM marks

    - ranks based on marks
    - ORDER BY marks DESC : says that rank them based on marks, highest will get 1, lowest will get 

# ranking based on branck by branch
    USE sql_learning;
    SELECT * , RANK() OVER(PARTITION BY BRANCH ORDER BY marks DESC) FROM marks



### DENSE RANK - Gives ranking for partition
* the same rank will be given for equal items Eg. if 2 students have marks 25, both will be ranked 3
* here, rank 4 will be given to the next highest marks holder
* the below code compares both dense and dense_rank

    USE sql_learning;
    SELECT * , 
    RANK() OVER(PARTITION BY BRANCH ORDER BY marks DESC),
    DENSE_RANK() OVER(PARTITION BY BRANCH ORDER BY marks DESC) 
    FROM marks

# get top 2 customers month wise who spent more money in that month / Find top 2 most paying customers of each month

    USE zomato;
    WITH rank_month AS (
        SELECT * FROM (
            SELECT monthname(date) AS 'month', user_id, SUM(amount) AS 'total',
            RANK() OVER(PARTITION BY MONTHNAME(date) ORDER BY SUM(amount) DESC) AS 'month_rank'
            FROM orders_subquery
            GROUP BY monthname(date), user_id
            ORDER BY month(date)
        ) t WHERE t.month_rank < 3)

    SELECT t1.month, t2.user_id, t1.month_rank, t2.name FROM rank_month t1
    JOIN users_subquery t2 ON t2.user_id = t1.user_id

# last_value 
    USE sql_learning;
    SELECT *, LAST_VALUE(name) OVER(ORDER BY marks DESC) FROM marks 

    - this query doesn't give expected result
    - CONCEPT of Frames
    - use unbounded following and unbounded preceding type to get expected answer as below

    USE sql_learning;
    SELECT *, LAST_VALUE(marks) 
    OVER(
        ORDER BY marks DESC 
        ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
    ) 
    FROM marks 

    # getting the marks last_value branch wise
        USE sql_learning;

# get the topper name, branch name and marks for each branch
    USE sql_learning;
    select name, branch, marks from ( 
        select *, first_value(name) 
	    over(partition by branch order by marks desc) as 'topper_name',
        first_value(marks) 
        over(partition by branch order by marks desc) as 'topper_marks'
    from marks) t
    where t.name = t.topper_name and t.marks = t.topper_marks

# last rank scorer of each branch
    USE sql_learning;
    select name, branch, marks from (
        select *, last_value(name) 
        over(partition by branch order by marks desc 
                ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) as 'topper_name',
        last_value(marks) 
        over(partition by branch order by marks desc
                ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) as 'topper_marks'
    from marks) t
    where t.name = t.topper_name and t.marks = t.topper_marks




############## LEAD and LAG functions ##############

LAG - gives the previous row value
* lag(data, 7) shows data which is 7 entry back
    USE sql_learning;
    select *, 
    lag(marks) over(order by student_id) 
    from marks

LEAD - gives the next row value
* lead(data, 7) shows data which is 7 entry forward
    USE sql_learning;
    select *, 
    lag(marks) over(order by student_id),
    lead(marks) over(order by student_id)
    from marks


# PARTITION BY branch
    USE sql_learning;
    select *, 
    lag(marks) over(partition by branch order by student_id),
    lead(marks) over(partition by branch order by student_id)
    from marks


# get the month on month revenue growth from zomato data
    USE zomato;
    select monthname(date), sum(amount), 
    (sum(amount) - lag(sum(amount)) over(order by month(date) asc))/
    lag(sum(amount)) over(order by month(date) asc) * 100 as 'MoM revenue'
    from orders_subquery
    group by monthname(date)
    order by month(date) asc

############### CUMULATIVE SUM ###############
It calculates the sum of set of values upto a given point in time and includes all previous values

# VK's career runs in 10th, 25th, 35th match
    use sql_learning;
    select * from (select 
    concat("Match-", CAST(row_number() OVER(ORDER BY ID) AS CHAR)) as 'match_num' ,
    sum(batsman_run) as 'runs_scored',
    sum(sum(batsman_run)) over(rows between unbounded preceding and current row) as 'career_runs'
    from ipl 
    where batter = 'V Kohli'
    group by ID) t
    where match_num = 'Match-10' or match_num = 'Match-25' or match_num = 'Match-35'

############### Quantiles ###############
It is a measure of distribution of a dataset that divides data into any number of equally sized interval
1. 10 equal parts: deciles
2. 4 equal parts: quartiles
3. 100 equal parts: percentiles

Percentile: each quantile represents a value below a certain percentage of data falls
eg: 25th percentile(1st quartile, Q1) represents value below which 25% data falls
eg: 50th percentile(Median) 

## Median of a data - 50th percentile
    USE sql_learning;
    select *, 
    percentile_disc(0.5) within group(order by marks) over() as 'median marks' 
    from marks

        - within groups is sorting data based on marks in ascending order

####### PERCENTILE CONT #######
    USE sql_learning;
    select *, 
    percentile_disc(0.5) within group(order by marks) over(partition by branch) as 'median marks disc',
    percentile_cont(0.5) within group(order by marks) over(partition by branch) as 'median marks cont' 
    from marks

# difference between percentile cont and percentile disc
    * percentile cont: calculates the continuous percentile values. It assumes that the data is distributed uniformly. This may give the value which may not be present in data
    * percentile cont: calculates the discreet percentile values. This always gives the result present in data

### Removing outlier user interquartile range 
    (Q3 - Q1) is IQR
    min value = (Q1 - 1.5(IQR))
    max value = (Q3 + 1.5(IQR))
    outliers are those points which are beyond min and max values

        select *
        from (select *, 
        percentile_disc(0.25) within group(order by marks) over() as 'q1',
        percentile_disc(0.75) within group(order by marks) over() as 'q3',
        (percentile_disc(0.75) within group(order by marks) over())-(percentile_disc(0.25) within group(order by marks) over()) as "IQR"
        from marks) t
        where t.marks > q1-1.5*IQR and t.marks < q3+1.5*IQR
        order by t.student_id








####### SEGMENTATION ####### 
It uses NTILE (it divides data into buckets on basis of some criteria or column) to divide dataset into equal sized groups based on some condition and performing calculations on each group saperately using window function

eg: classifying customers into premium, normal based on their purchase value

* this code creates 3 groups
    select *,
    NTILE(3) over() as 'groups'
    from marks

* this code creates 3 buckets based on marks - highest marks students in bucket 1, followed by bucket 2 and than bucket 3
    select *,
    NTILE(3) over(order by marks desc) as 'groups'
    from marks


* this code creates 3 buckets based on phone price and assignes name to those buckets 
    use sql_learning;
    select brand_name, model, price,
    CASE
        when bucket = 1 then 'budget'
        when bucket = 2 then 'mid-range'
        when bucket = 3 then 'premium'
    END AS 'phone_type'

    from (select brand_name, model, price,
    NTILE(3) over(order by price) as 'bucket'
    from smartphones) t






####### CUMULATIVE DISTRIBUTION ####### 
1. The cumulative distribution function is used to describe the probability distribution of random variables. 
2. It can be used to describe the probability for a discrete, continuous or mixed variable. 
3. It is obtained by summing up the probability density function and getting the cumulative probability for a random variable.

** It answers : what percentage of rows in the data have values less than or equal to current row

# those students above 90 percentile
    select * from (select * ,
    CUME_DIST() over(order by marks) as 'percentile_score'
    from marks) t
    where t.percentile_score > 0.90
