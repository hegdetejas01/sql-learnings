### Sorting based on single column

# Top 5 samsung phones with biggest screen size
    SELECT * FROM sql_learning.smartphones WHERE brand_name='samsung' ORDER BY screen_size DESC LIMIT 5

    - ORDER BY is used to SORT, 
    - DESC and ASC - ascending and descending
    - LIMIT 5 - outputs top 5 rows (head in pandas)


# sort all phones in descending order of number of total cameras
    SELECT model, (num_rear_cameras + num_front_cameras) AS 'total_cameras' FROM sql_learning.smartphones
    ORDER BY (num_rear_cameras + num_front_cameras) DESC

        ## OR ##
    SELECT model, (num_rear_cameras + num_front_cameras) AS 'total_cameras' FROM sql_learning.smartphones
    ORDER BY total_cameras DESC


# sort data based on ppi in decresing order - TOP 10
    SELECT model, 
    SQRT(resolution_width*resolution_width + resolution_height*resolution_height)/screen_size AS 'ppi' 
    FROM sql_learning.smartphones 
    ORDER BY ppi DESC LIMIT 10


# find the phone with 2nd highest battery capacity
    SELECT model, battery_capacity 
    FROM sql_learning.smartphones 
    ORDER BY battery_capacity DESC LIMIT 1,1

    - LIMIT takes 2 parameter (x,y) eg: LIMIT 1,1
    - that means, after x rows, give me y rows
    - for LIMIT 3,4 : it ignores row 0,1,2 and prints 4 rows starting from 3rd row [3,4,5,6] - start from 3rd row and give me 4 rows
    - for LIMIT 5,2 : it prints rows [5,6] - start from 5th row and give me 2 rows


# find the phone with 2nd lowest battery capacity

### group by single column

# group smartphones by brand and get the count of each brand - get the number of smartphones in each brand
    SELECT  brand_name, COUNT(*) AS 'total_phones' 
    FROM sql_learning.smartphones
    GROUP BY brand_name
    ORDER BY total_phones DESC

# group smartphones by brand and get the count of each brand - get the number of smartphones, average price, max rating, avg screen size, avg battery_capacity in each brand
    SELECT  brand_name, COUNT(model) AS 'total_phones', 
    ROUND(AVG(price),2) AS 'avg price', 
    ROUND(AVG(battery_capacity),2) AS 'avg battery', 
    ROUND(MAX(rating),2) AS 'max rating', 
    ROUND(AVG(screen_size),2) AS 'avg batter' 
    FROM sql_learning.smartphones
    GROUP BY brand_name
    ORDER BY total_phones DESC


# group smartphone by whether they have NFC and get avg price and avg rating
    SELECT has_nfc, AVG(price), AVG(rating) FROM sql_learning.smartphones GROUP BY has_nfc

# group smartphone by whether they have 5g and get avg price and avg rating
    SELECT has_5g, AVG(price), AVG(rating) FROM sql_learning.smartphones GROUP BY has_5g

# group smartphone by whether they have fast_charging_available and get avg price and avg rating
    SELECT fast_charging_available, AVG(price), AVG(rating) FROM sql_learning.smartphones GROUP BY fast_charging_available

# group smartphones by brand and find brands with highest number of models that have both NFC and IR blaster
    SELECT brand_name, COUNT(*) AS 'ans' FROM sql_learning.smartphones 
    WHERE has_ir_blaster = 'True' AND has_nfc = 'True'
    GROUP BY brand_name
    ORDER BY ans DESC LIMIT 1

    WHERE brand_name = 'samsung' AND has_5g = 'True'

#####################################################################################################
                                        HAVING CLAUSE
#####################################################################################################

* filtering the GROUP BY
* The way how,
    WHERE works on SELECT

    The same way,
    HAVING works on GROUP BY

* Applying condition for GROUP BY


# get the average price phones of those brands which has atleast 20 phones
    SELECT brand_name, ROUND(AVG(price),2) AS 'avg_price'
    FROM sql_learning.smartphones
    GROUP BY brand_name
    HAVING COUNT(*) > 20
    ORDER BY avg_price DESC

# avg rating of smartphone brands with more than 40 phones
    SELECT brand_name, ROUND(AVG(rating),2) AS 'avg_rating'
    FROM sql_learning.smartphones
    GROUP BY brand_name
    HAVING COUNT(*) > 40

# Top 3 brands with highest avg ram that have a refresh rate of at least 90 Hz and fast charging available and don't consider brands less than 10 phones
    SELECT brand_name, ROUND(AVG(ram_capacity)) AS 'avg_ram'
    FROM sql_learning.smartphones
    WHERE refresh_rate >= 90 AND fast_charging_available = 1
    GROUP BY brand_name
    HAVING COUNT(*) > 10
    ORDER BY avg_ram DESC LIMIT 3


# avg phone price of all phone brands with avg rating > 70 and number of phones > 10 among all 5g phones
    SELECT brand_name, ROUND(AVG(price)) AS 'avp' 
    FROM sql_learning.smartphones
    WHERE has_5g = 'True'
    GROUP BY brand_name
    HAVING count(*) > 10 AND avg(rating) > 70
