### JOIN OPERATIONS ###

# Cross Join on Groups(4 rows) and Users(6 rows)
    SELECT * FROM sql_learning.users_join_1 t1
    CROSS JOIN sql_learning.groups_join_1 t2

    - aliasing users table as t1 and groups table as t2
    - it has total of 24 rows


# Inner Join on users and membership
    - on user id column
    - in users - user id 8, 11 is present, which is not present in membership
    - in membership - user id 5, 6 is present, which is not present in users
    - therefore, all these details will not be present in result

    SELECT * FROM sql_learning.users_join_1 t1
    INNER JOIN sql_learning.membership_join_1 t2
    ON t1.user_id = t2.user_id

        ## OR ##
    SELECT * FROM sql_learning.users_join_1 t1
    JOIN sql_learning.membership_join_1 t2
    ON t1.user_id = t2.user_id


# Left Join on membership and users
    - membership: left table, users: right table
    - on user id column
    - in membership - user id 5, 6 is present, which is not present in users
    - in users - user id 8, 11 is present, which is not present in membership
    - therefore, user id 5,6 will also be in the result with the corresponding values of user table as NULL

    SELECT * FROM sql_learning.membership_join_1 t1
    LEFT JOIN sql_learning.users_join_1 t2
    ON t1.user_id = t2.user_id


# Right Join on membership and users
    - membership: left table, users: right table
    - on user id column
    - in membership - user id 5, 6 is present, which is not present in users
    - in users - user id 8, 11 is present, which is not present in membership
    - therefore, user id 8, 11 will also be in the result with the corresponding values of membership table as NULL

    SELECT * FROM sql_learning.membership_join_1 t1
    RIGHT JOIN sql_learning.users_join_1 t2
    ON t1.user_id = t2.user_id


# Full Outer Join - doesn't work in MySQL

    Other way to do FULL OUTER JOIN - left join union with right join
        SELECT * FROM sql_learning.membership_join_1 t1
        LEFT JOIN sql_learning.users_join_1 t2
        ON t1.user_id = t2.user_id
        UNION
        SELECT * FROM sql_learning.membership_join_1 t1
        RIGHT JOIN sql_learning.users_join_1 t2
        ON t1.user_id = t2.user_id


# Self Join.
    SELECT t1.name as 'name', t2.name as 'emergency'

    JOIN sql_learning.users_join_1 t2
    ON t1.emergency_contact = t2.user_id

##### Joining Multiple Tables #####

# I want name in the order_details table
    SELECT *
    FROM flipkart.order_details t1
    JOIN flipkart.orders t2
    ON t1.order_id = t2.order_id
    JOIN flipkart.users t3
    ON t2.user_id = t3.user_id

# Along with these, i also want category name
    SELECT t1.order_id, t2.user_id, t3.name, t4.category_id, t4.category, t3.state
    FROM flipkart.order_details t1
    JOIN flipkart.orders t2
    ON t1.order_id = t2.order_id
    JOIN flipkart.users t3
    ON t2.user_id = t3.user_id
    JOIN flipkart.category t4
    ON t4.category_id = t1.category_id

# order_id, name, city by joining users and orders
    SELECT t2.order_id, t1.name, t1.city
    FROM flipkart.users t1
    JOIN flipkart.orders t2
    ON t1.user_id = t2.user_id

# order_id and product category by joining order_details and category
    SELECT t1.order_id, t2.category
    FROM flipkart.order_details t1
    JOIN flipkart.category t2
    ON t1.category_id = t2.category_id



##### FILTERING ROWS #####

# find all orders placed in Pune
    SELECT * FROM flipkart.users t1
    JOIN flipkart.orders t2
    ON t1.user_id = t2.user_id
    WHERE t1.city = 'Pune'

# find all orders under chair vertical
    SELECT * FROM flipkart.order_details t1
    JOIN flipkart.category t2
    ON t1.category_id = t2.category_id
    WHERE t2.vertical = 'Chairs'




# find all profitable orders
    SELECT t1.order_id, t2.amount, sum(t2.profit) AS 'net_pr'
    FROM flipkart.orders t1
    JOIN flipkart.order_details t2
    ON t2.order_id = t1.order_id
    GROUP BY order_id
    HAVING netPr>0

# Find customer who has placed maximum orders
    SELECT t2.name, COUNT(*) AS 'total_orders' 
    FROM flipkart.orders t1
    JOIN flipkart.users t2 
    ON t1.user_id = t2.user_id
    GROUP BY t2.name
    ORDER BY total_orders DESC LIMIT 1

# which is the most profitable vertical
    SELECT t2.vertical, SUM(t1.profit) as 'net_pr'
    FROM flipkart.order_details t1
    JOIN flipkart.category t2 
    ON t1.category_id = t2.category_id
    GROUP BY t2.vertical
    ORDER BY net_pr DESC LIMIT 1

# which is the most profitable state
    SELECT t3.state, SUM(t1.profit) AS 'net_pr' 
    FROM flipkart.order_details t1
    JOIN flipkart.orders t2 ON t1.order_id = t2.order_id
    JOIN flipkart.users t3 ON t2.user_id = t3.user_id
    GROUP BY t3.state
    ORDER BY net_pr DESC LIMIT 1

# find all category with profit higher than 2000
    SELECT t2.vertical, SUM(t1.profit) as 'net_pr'
    FROM flipkart.order_details t1
    JOIN flipkart.category t2 
    ON t1.category_id = t2.category_id
    GROUP BY t2.vertical
    HAVING net_pr > 2000
    ORDER BY net_pr DESC
